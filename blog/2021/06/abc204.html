    #include <bits/stdc++.h>
using namespace std;
using ull = unsigned long long;
using ull1d = vector<ull>;
using ull2d = vector<ull1d>;
#define loop(i,n) for(ull i=0;i<n;i++)

vector<bool> seen;
ull2d Graph;
ull c = 0;

void dfs(ull v) {
  seen[v] = true;
  c++;
  for (ull next_v : Graph[v]) {
    if(seen[next_v]) continue;
    dfs(next_v);
  }
}

int main(){
  ull n,m;
  cin >> n >> m;
  seen.resize(n);
  Graph.resize(n);
  loop(i,m){
    ull a,b;
    cin >> a >> b;
    Graph[a-1].push_back(b - 1);
  }
  loop(i,n){
    seen.assign(n,false);
    dfs(i);
  }
  cout << c << endl;
  return 0;
}
