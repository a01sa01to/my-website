    #include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef vector<ll> ll1d;
#define loop(i, n) for (ll i = 0; i < n; i++)
#define all(vec) vec.begin(), vec.end()

int main() {
  cout << fixed << setprecision(15);
  ll n, m;
  cin >> n >> m;
  ll1d a(n), b(m);
  loop(i, n) { cin >> a[i]; }
  loop(i, m) { cin >> b[i]; }
  sort(all(a));
  sort(all(b));
  ll minim = 1e9;
  ll i = 0, j = 0;
  while (i < n && j < m) {
    if (a[i] - b[j] >= 0) {
      minim = min(minim, a[i] - b[j]);
      j++;
    }
    else {
      minim = min(minim, b[j] - a[i]);
      i++;
    }
  }
  cout << minim << endl;
  return 0;
}
